<h2>Crypto Tax Calculator</h2>

<div>

  <label>Buy Price ($): <input type="number" id="buyPrice" /></label><br><br>

  <label>Sell Price ($): <input type="number" id="sellPrice" /></label><br><br>

  <label>Number of Coins: <input type="number" id="coins" /></label><br><br>

  <label>Holding Period (days): <input type="number" id="holdingPeriod" /></label><br><br>

  <button onclick="calculateTax()">Calculate Tax</button>

</div>

<h3 id="result"></h3>

<script>

function calculateTax() {

    const buyPrice = parseFloat(document.getElementById("buyPrice").value);

    const sellPrice = parseFloat(document.getElementById("sellPrice").value);

    const coins = parseFloat(document.getElementById("coins").value);

    const holdingPeriod = parseInt(document.getElementById("holdingPeriod").value);

    if (isNaN(buyPrice) || isNaN(sellPrice) || isNaN(coins) || isNaN(holdingPeriod)) {

        document.getElementById("result").innerText = "Please fill in all fields correctly.";

        return;

    }

    const totalCost = buyPrice * coins;

    const totalRevenue = sellPrice * coins;

    const gain = totalRevenue - totalCost;

    let taxRate = holdingPeriod > 365 ? 0.15 : 0.30;  // Example: 15% long-term, 30% short-term

    const taxOwed = gain > 0 ? gain * taxRate : 0;

    document.getElementById("result").innerText = `

      Gain/Loss: $${gain.toFixed(2)}

      Estimated Tax Rate: ${(taxRate * 100)}%

      Estimated Tax Owed: $${taxOwed.toFixed(2)}

    `;

}

</script>
